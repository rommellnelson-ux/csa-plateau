1. CSP Header Validation

const meta = document.querySelector('meta[http-equiv="Content-Security-Policy"]'); if(meta) { console.log('PASS: CSP header found'); console.log('Policy:', meta.getAttribute('content')); } else { console.log('FAIL: CSP header not found'); }

2. Encryption Utility Test

const testData = {patient_id: '123', name: 'Test'}; const encrypted = SecureStorage.encrypt(testData); const decrypted = SecureStorage.decrypt(encrypted); console.log('Test data:', testData); console.log('Encrypted:', encrypted); console.log('Decrypted:', decrypted); console.log('Match:', JSON.stringify(testData) === JSON.stringify(decrypted) ? 'PASS' : 'FAIL');

3. localStorage Cleanup Test

const keys = ['csa2_patients', 'csa2_consultations', 'csa2_soins']; keys.forEach(k => localStorage.setItem(k, 'test-value')); console.log('Keys before clear:', keys.map(k => localStorage.getItem(k))); // Simulate logout cleanup keys.forEach(k => localStorage.removeItem(k)); console.log('Keys after clear:', keys.map(k => localStorage.getItem(k))); console.log('Cleanup:', keys.every(k => !localStorage.getItem(k)) ? 'PASS' : 'FAIL');

4. XSS Protection Test

// Test that XSS payloads are properly escaped const payload = ''; const escaped = escHtml(payload); console.log('XSS Payload:', payload); console.log('Escaped:', escaped); console.log('Contains dangerous chars:', /[<>]/.test(escaped) ? 'FAIL' : 'PASS');

5. Security Test Suite (CSPTest)
